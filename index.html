1. A Tranformada Discreta de Fourier

O seguinte programa calcula o espectro de uma imagem com uma senoide de 8 per√≠odos, a fim de analisar seu comportamento no dom√≠nio da frequ√™ncia:

1.1 transformada de Fourier da sen√≥ide (PNG)

DFT.py
import cv2
import numpy as np

def swap_quadrants(image):
    # Se a imagem tiver dimens√µes √≠mpares, corta para o tamanho par mais pr√≥ximo
    image = image[:image.shape[0] & -2, :image.shape[1] & -2]

    center_x = image.shape[1] // 2
    center_y = image.shape[0] // 2

    # Rearranja os quadrantes
    A = image[:center_y, :center_x]
    B = image[:center_y, center_x:]
    C = image[center_y:, :center_x]
    D = image[center_y:, center_x:]

    # Troca os quadrantes (Topo-Esquerda com Baixo-Direita)
    tmp = np.copy(A)
    A[:], D[:] = D, tmp

    # Troca o quadrante (Topo-Direita com Baixo-Esquerda)
    tmp = np.copy(C)
    C[:], B[:] = B, tmp

def main():
    # L√™ a imagem em escala de cinza
    image = cv2.imread("senoide.png", cv2.IMREAD_GRAYSCALE)
    if image is None:
        print("Erro ao abrir a imagem")
        return

    # Expande a imagem para o tamanho √≥timo para a transformada de Fourier
    dft_M = cv2.getOptimalDFTSize(image.shape[0])
    dft_N = cv2.getOptimalDFTSize(image.shape[1])
    padded = cv2.copyMakeBorder(image, 0, dft_M - image.shape[0], 0, dft_N - image.shape[1], cv2.BORDER_CONSTANT, value=0)

    # Prepara a matriz complexa
    complex_image = cv2.merge([padded.astype(float), np.zeros_like(padded, dtype=float)])

    # Calcula a Transformada de Fourier Discreta (DFT)
    complex_image = cv2.dft(complex_image)
    swap_quadrants(complex_image)

    # Separa a imagem complexa em partes real e imagin√°ria
    planos = cv2.split(complex_image)

    # Calcula magnitude e fase
    magn, fase = cv2.cartToPolar(planos[0], planos[1], angleInDegrees=False)
    fase = cv2.normalize(fase, None, 0, 1, cv2.NORM_MINMAX)

    # Calcula o log da magnitude
    magn += 1
    magn = np.log(magn)
    magn = cv2.normalize(magn, None, 0, 1, cv2.NORM_MINMAX)

    # Exibe as imagens processadas
    cv2.imshow("Imagem Original", image)
    cv2.imshow("Espectro de Magnitude", magn)
    cv2.imshow("Espectro de Fase", fase)

    cv2.waitKey(0)
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()

2.Filtragem no Dom√≠nio da Frequ√™ncia

Ao empregar o programa exemplos/dftfilter.cpp como ponto de partida, desenvolvi uma implementa√ß√£o do filtro homom√≥rfico com o objetivo de aprimorar imagens afetadas por ilumina√ß√£o irregular. Para realizar essa tarefa, criei uma cena propositalmente mal iluminada, visando simular condi√ß√µes desafiadoras. Ajustei os par√¢metros do filtro homom√≥rfico de maneira criteriosa para corrigir as irregularidades de ilumina√ß√£o de forma otimizada

homomorfico.py
import cv2
import numpy as np

def homomorphic_filter(image, cutoff_freq_low, cutoff_freq_high, gamma_low, gamma_high):
    # Converter a imagem para float32
    image_float32 = np.float32(image)

    # Aplicar a transforma√ß√£o logar√≠tmica
    log_image = np.log1p(image_float32)

    # Realizar a Transformada de Fourier (FFT)
    fft_image = np.fft.fft2(log_image)

    # Criar o filtro
    rows, cols = image.shape
    filtro = np.zeros((rows, cols), dtype=np.float32)

    for i in range(rows):
        for j in range(cols):
            d = np.sqrt((i - rows/2)**2 + (j - cols/2)**2)
            filtro[i, j] = (gamma_high - gamma_low) * (1 - np.exp(-cutoff_freq_low * (d**2) / (cutoff_freq_high - cutoff_freq_low))) + gamma_low

    # Aplicar o filtro no dom√≠nio da frequ√™ncia
    fft_filtrada = fft_image * filtro

    # FFT inversa
    inversa_fft = np.fft.ifft2(fft_filtrada)

    # Transforma√ß√£o exponencial
    imagem_exp = np.expm1(np.real(inversa_fft))

    # Normalizar para exibi√ß√£o
    resultado = cv2.normalize(imagem_exp, None, 0, 255, cv2.NORM_MINMAX)
    resultado = np.uint8(resultado)

    return resultado

if __name__ == "__main__":
    # Carregar a imagem
    imagem = cv2.imread("homomorfica.png", cv2.IMREAD_GRAYSCALE)

    # Aplicar o filtro homom√≥rfico
    imagem_filtrada = homomorphic_filter(imagem, cutoff_freq_low=0.1, cutoff_freq_high=30, gamma_low=25, gamma_high=1)

    # Exibir o resultado
    cv2.imshow("Filtro Homom√≥rfico", imagem_filtrada)
    cv2.imshow("Original", imagem)
    cv2.waitKey(0)
    cv2.destroyAllWindows()

O filtro √© projetado para controlar a influ√™ncia de diferentes frequ√™ncias na imagem resultante. Ap√≥s a aplica√ß√£o do filtro, √© realizada a inversa da FFT e uma transforma√ß√£o exponencial para recuperar a imagem original. 3.Segmenta√ß√£o de imagens (cannypoints.py)

Ao implementar o programa cannypoints.py, tomei como refer√™ncia os c√≥digos de exemplo canny.py e pontilhismo.py. A proposta foi integrar as caracter√≠sticas distintivas do algoritmo de detec√ß√£o de bordas Canny para aprimorar a qualidade da imagem pontilhista resultante. Na minha abordagem, optei por utilizar a informa√ß√£o das bordas detectadas pelo algoritmo de Canny de maneira espec√≠fica, utilizando as posi√ß√µes dos pixels de borda identificados para desenhar pontos nos locais correspondentes na imagem gerada pelo programa em Python.

cannypoints.py
import numpy as np
import random
import cv2  # √â necess√°rio importar a biblioteca OpenCV

STEP = 3  # Tamanho do passo para percorrer as bordas
JITTER = 3  # Valor de ajuste aleat√≥rio para posicionar os pontos de maneira irregular
RAIO = 10  # Raio dos c√≠rculos a serem desenhados

# Defina os par√¢metros manualmente
top_slider = 5  # Ajuste manualmente
TrackbarName = "Threshold inferior"

image = cv2.imread("lena.jpg", cv2.IMREAD_GRAYSCALE)

if image is None:
    print("Could not open or find the image")
else:
    height, width = image.shape

    # Utilizando as bordas para desenhar pontos na imagem gerada
    points = np.full_like(image, 255, dtype=np.uint8)  # Inicializa uma imagem com fundo branco

    edges = cv2.Canny(image, top_slider, 3 * top_slider)  # Aplica o detector de bordas de Canny

    yrange, xrange = np.where(edges > 0)  # Obt√©m as coordenadas dos pixels de borda

    for x, y in zip(xrange, yrange):
        for _ in range(STEP):
            # Adiciona um componente aleat√≥rio para posicionar os pontos de maneira irregular
            xi = x + random.randint(-JITTER, JITTER + 1)
            yi = y + random.randint(-JITTER, JITTER + 1)

            # Garante que os pontos n√£o ultrapassem os limites da imagem
            xi = max(0, min(xi, width - 1))
            yi = max(0, min(yi, height - 1))

            # Obt√©m o valor de intensidade do pixel para definir a cor do c√≠rculo
            gray = int(image[yi, xi])

            # Desenha um c√≠rculo na imagem de pontos
            cv2.circle(points, (xi, yi), RAIO, (gray, gray, gray), thickness=cv2.FILLED, lineType=cv2.LINE_AA)

    # Exibe a imagem resultante
    cv2.imshow("Pontos", points)
    cv2.waitKey(0)

    # Salva a imagem resultante
    cv2.imwrite("pontos_com_bordas.png", points)

# Fecha as janelas abertas
cv2.destroyAllWindows()

Este c√≥digo Python utiliza a biblioteca OpenCV para realizar uma opera√ß√£o de pontilhismo din√¢mico em uma imagem em tons de cinza. Primeiramente, o programa l√™ uma imagem, como exemplificado pela imagem "lena.jpg". Em seguida, aplica o detector de bordas de Canny para identificar regi√µes de alta varia√ß√£o de intensidade, representando bordas na imagem original. As coordenadas desses pixels de borda s√£o ent√£o utilizadas para posicionar c√≠rculos na imagem resultante. Cada ponto √© colocado com um deslocamento aleat√≥rio (jitter) para evitar uma disposi√ß√£o regular e conferir um car√°ter mais org√¢nico √† representa√ß√£o. A intensidade dos pixels na imagem original determina a cor dos c√≠rculos, onde a sobreposi√ß√£o de c√≠rculos destaca as caracter√≠sticas de borda. Uma observa√ß√£o: deixei a sensibilidade de canny alta, pois quanto menos bordas forem encontradas menos pontos s√£o colocados, uma boa solu√ß√£o seria aumentar o raio das bordas nesse caso.

4.Quantiza√ß√£o vetorial com k-means

Ao abordar a implementa√ß√£o do algoritmo K-Means, optei por realizar 10 rodadas distintas do algoritmo, especificando a inicializa√ß√£o dos centros de forma aleat√≥ria atrav√©s do par√¢metro KMEANS_RANDOM_CENTERS. Essa escolha visa explorar a influ√™ncia da inicializa√ß√£o na estabilidade e consist√™ncia dos agrupamentos:

k-means.py
import cv2
import numpy as np

# N√∫mero de clusters desejados
nClusters = 8

# N√∫mero de rodadas do algoritmo K-Means
nRodadas = 1

# Carregar a imagem

# Reformatar a imagem para um vetor de amostras

# Crit√©rios de parada para o algoritmo K-Means
criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 10000, 0.0001)

# Inicializar centros de forma aleat√≥ria
flags = cv2.KMEANS_RANDOM_CENTERS
_, labels, centers = cv2.kmeans(samples, nClusters, None, criteria, nRodadas, flags)

# Criar imagem rotulada com base nos centros encontrados
rotulada = centers[labels].reshape(img.shape).astype(np.uint8)

# Exibir a imagem resultante da primeira rodada e salvar
cv2.imshow("kmeans", rotulada)
cv2.imwrite("saida_random_centers_rodada_1.jpg", rotulada)
cv2.waitKey(0)

# Realizar 9 rodadas adicionais
for rodada in range(2, 11):
    _, labels, centers = cv2.kmeans(samples, nClusters, None, criteria, nRodadas, flags)
    rotulada = centers[labels].reshape(img.shape).astype(np.uint8)
    cv2.imwrite(f"saida_random_centers_rodada_{rodada}.jpg", rotulada)

# Fechar as janelas abertas
cv2.destroyAllWindows()

Inicialmente, a imagem √© convertida em um vetor de amostras, onde cada amostra corresponde a um pixel com tr√™s componentes de cor. O algoritmo K-Means √© ent√£o empregado com a configura√ß√£o de oito clusters e uma √∫nica rodada (nRodadas=1).A inicializa√ß√£o dos centros √© realizada de forma aleat√≥ria (KMEANS_RANDOM_CENTERS). A imagem resultante da primeira itera√ß√£o √© exibida e salva. Em seguida, o algoritmo √© executado em mais nove rodadas, cada uma gerando uma imagem rotulada e salva.

Ao longo das m√∫ltiplas rodadas, a variabilidade nas imagens resultantes pode ser observada devido √† inicializa√ß√£o aleat√≥ria dos centros. Isso destaca a influ√™ncia significativa que as condi√ß√µes iniciais t√™m no processo de agrupamento. Cada imagem captura diferentes nuances na segmenta√ß√£o de cores, revelando a sensibilidade do algoritmo K-Means √† escolha inicial dos centros. O uso de m√∫ltiplas rodadas proporciona uma perspectiva mais abrangente dos agrupamentos poss√≠veis, destacando a natureza estoc√°stica do algoritmo e a necessidade de avaliar sua robustez frente a diferentes condi√ß√µes iniciais.

5.Filtragem de forma com morfologia matem√°tica

Para superar desafios no reconhecimento de caracteres em visores segmentados, prop√µe-se um programa baseado em opera√ß√µes morfol√≥gicas, utilizando morfologia.cpp. O objetivo √© otimizar o reconhecimento de d√≠gitos. A preserva√ß√£o da separa√ß√£o do ponto decimal √© crucial para evitar equ√≠vocos no reconhecimento num√©rico.

correcaoDigito.py
import cv2
import numpy as np

def pre_processamento(image_path):
    # Carregar a imagem
    image = cv2.imread(image_path, cv2.IMREAD_UNCHANGED)

    if image is None:
        print("Erro ao carregar a imagem:", image_path)
        return None

    # Elemento estruturante
    str_element = cv2.getStructuringElement(cv2.MORPH_RECT, (1, 10))

    # Eros√£o para unir os segmentos
    eroded = cv2.erode(image, str_element)

    # Dilata√ß√£o para recuperar a forma original, mantendo os segmentos unidos
    dilated = cv2.dilate(eroded, str_element)

    # Exibi√ß√£o da imagem de pr√©-processamento
    cv2.imshow("Pr√©-processamento", dilated)
    cv2.waitKey(0)

    return dilated

# Exemplo de uso
imagem_pre_processada = pre_processamento("digitos-2.png")

O elemento estruturante (str_element), definido como um ret√¢ngulo de dimens√µes (1, 10), √© projetado para unir segmentos verticalmente, sendo uma dimens√£o estrat√©gica para preservar a separa√ß√£o horizontal entre os d√≠gitos. Essa escolha espec√≠fica (1, 10) √© crucial para evitar a uni√£o do ponto decimal com os d√≠gitos adjacentes, garantindo a correta interpreta√ß√£o num√©rica. A sequ√™ncia de opera√ß√µes, eros√£o e dilata√ß√£o, √© executada de maneira a unir os segmentos do d√≠gito verticalmente, contribuindo para a coes√£o do caractere reconhecido.
